# Convert Timestamp into categorical feature of Time_of_day
# 1 = morning, 2 = Mid-day, 3 = Evening
arr = []
for i in data.timestamp:
    hour = i.hour
    if (4<hour<=12):
        # Morning
        arr.append(1)
    elif (12<hour<=20):
        # Mid-day
        arr.append(2)
    else:
        # Evening
        arr.append(3)
data['Time_of_day'] = arr
data = data.drop('timestamp',axis=1)
